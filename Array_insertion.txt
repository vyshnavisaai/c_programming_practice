#include <stdio.h>

int add(int n,int capacity,int key,int a[])
{
    if(n>=capacity)
        return n;
    a[n]=key;
    return n+1;
    
}
int main()
{
    int i,left,right;
    int a[20]={1,3,5,7,9,11};
    int capacity=sizeof(a)/sizeof(a[0]);
    int n=6;
    printf("before insertion :");
    for(i=0;i<n;i++)
    {
        printf("%d  ",a[i]);
    }
    
    int key=20;
    n= add(n,capacity,key,a);
    printf("\nafter insertion  :");

    for(i=0;i<n;i++)
    {
     printf("%d  ",a[i]);
    }
    
}
